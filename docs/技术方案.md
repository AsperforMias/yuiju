# 带参数原子行为的技术方案

## 问题描述
"吃 xx"、"买 xx"这样的原子行为，有参数，应该如何让模型处理。

## 技术方案：多 Agent 架构

### 核心思路

使用多 Agent 架构来处理参数化行为：
1. **主 Agent**：负责选择行为类型（如 "eat_item"）
2. **参数 Agent**：每种参数化行为有专门的 Agent 负责选择具体参数
3. **Function Tools**：为主 Agent 提供按需查询能力

### 工作流程

```
1. 主 Agent 看到可选行为：["wake_up", "eat_item", "buy_item", "sleep"]
2. 主 Agent 可以调用查询工具了解详情：queryAvailableFood()
3. 主 Agent 基于查询结果选择行为："eat_item"
4. 系统检测到参数化行为，调用对应的参数 Agent
5. EatItem Agent 根据当前状态选择具体物品："苹果"
6. 执行最终行为：eat_item("苹果")
```

export enum ActionId {
  // ... 现有行为

  Buy_Item = 'buy_item',

export interface ActionMetadata {
  action: ActionId;
  description: string;
  precondition: (context: ActionContext) => boolean | Promise<boolean>;
  
  // 参数选择 Agent
  parameterAgent?: (context: ActionContext) => Promise<Array<ActionParameter>>;
  
  // 执行器接收参数
  executor: (context: ActionContext, parameter?: Array<ActionParameter>) => void | Promise<void>;
  durationMin: number | ((context: ActionContext, parameter?: Array<ActionParameter>) => Promise<number>);
  completionEvent?: string | ((context: ActionContext, parameter?: Array<ActionParameter>) => string | Promise<string>);
}

// 参数接口
export interface ActionParameter {
  value: string;           // 参数值，如："苹果"
  description?: string;    // 参数描述，如："苹果可以恢复10点体力"
  metadata?: Record<string, any>; // 额外信息，如：{ price: 5, stamina: 20 }
}

export async function chooseAction(
  actionList: ActionMetadata[],
  context: ActionContext,
  actionMemoryList: ActionMemory[]

        reason: z.string().describe('选择action的原因'),
        durationMinute: z.number().optional().describe('action的持续时间（分钟）'),
        updateShortTermPlan: z.array(z.string()).optional(),
        updateLongTermPlan: z.string().optional(),

    }),
    prompt: chooseActionPrompt({
      actionList,
      context,
      actionMemoryList,
      availableTools: Object.keys(actionQueryTools) // 告知可用工具
    }),
  });

  return output;
}

## 4. 参数 Agent 实现

```typescript
// EatItem 专用参数 Agent
export async function chooseEatItemParameter(context: ActionContext): Promise<Array<ActionParameter>> {
  const availableFood = await getAvailableFood(context);
  
  if (availableFood.length === 0) {
    throw new Error('没有可用的食物');
  }

  // 返回所有可用的食物选项，让后续的选择逻辑处理
  return availableFood.map(food => ({
    value: food.name,
    description: `${food.name}可以恢复${food.stamina}点体力`,
    metadata: {
      stamina: food.stamina,
      category: 'food'
    }
  }));
}

// BuyItem 专用参数 Agent
export async function chooseBuyItemParameter(context: ActionContext): Promise<Array<ActionParameter>> {
  const availableItems = await getAvailableItems(context);
  const money = context.charactorState.money;
  
  // 过滤买得起的物品
  const affordableItems = availableItems.filter(item => item.price <= money);
  
  if (affordableItems.length === 0) {
    throw new Error('没有买得起的物品');
  }

  // 返回所有买得起的物品选项
  return affordableItems.map(item => ({
    value: item.name,
    description: `${item.name}需要${item.price}金币`,
    metadata: {
      price: item.price,
      category: item.category
    }
  }));
}
```

## 5. 参数化行为定义

```typescript
// 参数化行为定义
export const parameterizedActions: ActionMetadata[] = [
  {
    action: ActionId.Eat_Item,
    description: '吃食物恢复体力（可以调用 queryAvailableFood 查看可用食物）',
    
    precondition: (context) => context.charactorState.stamina < 90,
    
    parameterAgent: chooseEatItemParameter,
    
    async executor(context, parameters) {
      // parameters 是可选的参数数组，主 Agent 会从中选择一个
      if (!parameters || parameters.length === 0) {
        throw new Error('没有可用的食物参数');
      }
      
      // 这里可以实现选择逻辑，或者由主 Agent 在调用时已经选定了具体参数
      const selectedFood = parameters[0]; // 简化处理，取第一个参数
      
      await context.charactorState.setAction(ActionId.Eat_Item);
      
      const stamina = selectedFood.metadata?.stamina || 10;
      await context.charactorState.changeStamina(stamina);
      await context.charactorState.consumeItem(selectedFood.value);
    },
    
    durationMin: 10,
  },
  
  {
    action: ActionId.Buy_Item,
    description: '购买物品（可以调用 queryAvailableItems 查看可购买物品）',
    
    precondition: (context) => context.charactorState.money > 0,
    
    parameterAgent: chooseBuyItemParameter,
    
    async executor(context, parameters) {
      if (!parameters || parameters.length === 0) {
        throw new Error('没有可用的购买参数');
      }
      
      const selectedItem = parameters[0]; // 简化处理，取第一个参数
      
      await context.charactorState.setAction(ActionId.Buy_Item);
      
      const price = selectedItem.metadata?.price || 5;
      await context.charactorState.changeMoney(-price);
      await context.charactorState.addItem(selectedItem.value);
    },
    
    durationMin: 5,
  }
];
```

## 6. 主流程集成

```typescript
// 修改主执行流程
export async function executeAction(
  selectedAction: ActionId,
  context: ActionContext
): Promise<string> {
  const actionMeta = getActionMetadata(selectedAction);
  
  if (actionMeta.parameterAgent) {
    // 参数化行为：先获取可用参数，再执行
    const availableParameters = await actionMeta.parameterAgent(context);
    
    if (availableParameters.length === 0) {
      throw new Error(`没有可用的参数执行 ${selectedAction}`);
    }
    
    // 这里可以实现参数选择逻辑
    // 方案1：简单取第一个参数
    // 方案2：再次调用 LLM 从参数中选择
    // 方案3：根据某种策略自动选择最优参数
    
    const selectedParameter = availableParameters[0]; // 简化处理
    
    await actionMeta.executor(context, [selectedParameter]);
    
    if (actionMeta.completionEvent) {
      return typeof actionMeta.completionEvent === 'function'
        ? await actionMeta.completionEvent(context, [selectedParameter])
        : actionMeta.completionEvent;
    }
  } else {
    // 普通行为：直接执行
    await actionMeta.executor(context);
    
    if (actionMeta.completionEvent) {
      return typeof actionMeta.completionEvent === 'function'
        ? await actionMeta.completionEvent(context)
        : actionMeta.completionEvent;
    }
  }
  
  return '';
}

// 或者更完整的参数选择流程
export async function executeActionWithParameterSelection(
  selectedAction: ActionId,
  context: ActionContext
): Promise<string> {
  const actionMeta = getActionMetadata(selectedAction);
  
  if (actionMeta.parameterAgent) {
    // 1. 获取所有可用参数
    const availableParameters = await actionMeta.parameterAgent(context);
    
    if (availableParameters.length === 0) {
      throw new Error(`没有可用的参数执行 ${selectedAction}`);
    }
    
    // 2. 如果只有一个参数，直接使用
    if (availableParameters.length === 1) {
      const selectedParameter = availableParameters[0];
      await actionMeta.executor(context, [selectedParameter]);
      
      if (actionMeta.completionEvent) {
        return typeof actionMeta.completionEvent === 'function'
          ? await actionMeta.completionEvent(context, [selectedParameter])
          : actionMeta.completionEvent;
      }
    } else {
      // 3. 多个参数时，调用 LLM 选择最合适的
      const selectedParameter = await selectBestParameter(
        selectedAction,
        availableParameters,
        context
      );
      
      await actionMeta.executor(context, [selectedParameter]);
      
      if (actionMeta.completionEvent) {
        return typeof actionMeta.completionEvent === 'function'
          ? await actionMeta.completionEvent(context, [selectedParameter])
          : actionMeta.completionEvent;
      }
    }
  } else {
    // 普通行为：直接执行
    await actionMeta.executor(context);
    
    if (actionMeta.completionEvent) {
      return typeof actionMeta.completionEvent === 'function'
        ? await actionMeta.completionEvent(context)
        : actionMeta.completionEvent;
    }
  }
  
  return '';
}

// 参数选择 Agent
async function selectBestParameter(
  actionId: ActionId,
  availableParameters: ActionParameter[],
  context: ActionContext
): Promise<ActionParameter> {
  const prompt = `
你是悠酱，现在要执行 ${actionId} 行为。请从以下选项中选择最合适的：

${availableParameters.map((param, index) => 
  `${index + 1}. ${param.description || param.value}`
).join('\n')}

当前状态：
- 体力：${context.charactorState.stamina}/100
- 金币：${context.charactorState.money}
- 时间：${context.worldState.time.format('HH:mm')}

请选择最合适的选项。
  `;

  const { output } = await generateText({
    model: siliconflow.chat('Qwen/Qwen3-8B'),
    output: Output.object({
      schema: z.object({
        selectedIndex: z.number().min(0).max(availableParameters.length - 1).describe('选择的选项索引（从0开始）'),
        reason: z.string().describe('选择理由')
      })
    }),
    prompt
  });

  return availableParameters[output.selectedIndex];
}
```

3. **创建参数 Agent**：为每种参数化行为实现专门的参数选择 Agent

## 9. 优势

- **职责分离**：主 Agent 专注行为选择，参数 Agent 专注具体参数
- **按需查询**：主 Agent 可以主动获取信息辅助决策
- **高度模块化**：每种参数化行为有独立的处理逻辑
- **易于扩展**：添加新的参数化行为只需实现对应的参数 Agent
- **调试友好**：可以单独测试和优化每个组件
- **清晰的数据流**：信息流向明确，便于理解和维护
