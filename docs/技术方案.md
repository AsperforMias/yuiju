# 带参数原子行为的技术方案

## 问题描述
"吃 xx"、"买 xx"这样的原子行为，有参数，应该如何让模型处理。

## 技术方案：多 Agent 架构

### 核心思路

使用多 Agent 架构来处理参数化行为：
1. **主 Agent**：负责选择行为类型（如 "eat_item"）
2. **参数 Agent**：每种参数化行为有专门的 Agent 负责选择具体参数
3. **Function Tools**：为主 Agent 提供按需查询能力

### 工作流程

```
1. 主 Agent 看到可选行为：["wake_up", "eat_item", "buy_item", "sleep"]
2. 主 Agent 可以调用查询工具了解详情：queryAvailableFood()
3. 主 Agent 基于查询结果选择行为："eat_item"
4. 系统检测到参数化行为，调用对应的参数 Agent
5. EatItem Agent 根据当前状态选择具体物品："苹果"
6. 执行最终行为：eat_item("苹果")
```

```typescript
// 修改主执行流程
export async function executeAction(
  selectedAction: ActionId,
  context: ActionContext
): Promise<string> {
  const actionMeta = getActionMetadata(selectedAction);
  
  if (actionMeta.parameterAgent) {
    // 参数化行为：先获取可用参数，再执行
    const availableParameters = await actionMeta.parameterAgent(context);
    
    if (availableParameters.length === 0) {
      throw new Error(`没有可用的参数执行 ${selectedAction}`);
    }
    
    // 这里可以实现参数选择逻辑
    // 方案1：简单取第一个参数
    // 方案2：再次调用 LLM 从参数中选择
    // 方案3：根据某种策略自动选择最优参数
    
    const selectedParameter = availableParameters[0]; // 简化处理
    
    await actionMeta.executor(context, [selectedParameter]);
    
    if (actionMeta.completionEvent) {
      return typeof actionMeta.completionEvent === 'function'
        ? await actionMeta.completionEvent(context, [selectedParameter])
        : actionMeta.completionEvent;
    }
  } else {
    // 普通行为：直接执行
    await actionMeta.executor(context);
    
    if (actionMeta.completionEvent) {
      return typeof actionMeta.completionEvent === 'function'
        ? await actionMeta.completionEvent(context)
        : actionMeta.completionEvent;
    }
  }
  
  return '';
}

// 或者更完整的参数选择流程
export async function executeActionWithParameterSelection(
  selectedAction: ActionId,
  context: ActionContext
): Promise<string> {
  const actionMeta = getActionMetadata(selectedAction);
  
  if (actionMeta.parameterAgent) {
    // 1. 获取所有可用参数
    const availableParameters = await actionMeta.parameterAgent(context);
    
    if (availableParameters.length === 0) {
      throw new Error(`没有可用的参数执行 ${selectedAction}`);
    }
    
    // 2. 如果只有一个参数，直接使用
    if (availableParameters.length === 1) {
      const selectedParameter = availableParameters[0];
      await actionMeta.executor(context, [selectedParameter]);
      
      if (actionMeta.completionEvent) {
        return typeof actionMeta.completionEvent === 'function'
          ? await actionMeta.completionEvent(context, [selectedParameter])
          : actionMeta.completionEvent;
      }
    } else {
      // 3. 多个参数时，调用 LLM 选择最合适的
      const selectedParameter = await selectBestParameter(
        selectedAction,
        availableParameters,
        context
      );
      
      await actionMeta.executor(context, [selectedParameter]);
      
      if (actionMeta.completionEvent) {
        return typeof actionMeta.completionEvent === 'function'
          ? await actionMeta.completionEvent(context, [selectedParameter])
          : actionMeta.completionEvent;
      }
    }
  } else {
    // 普通行为：直接执行
    await actionMeta.executor(context);
    
    if (actionMeta.completionEvent) {
      return typeof actionMeta.completionEvent === 'function'
        ? await actionMeta.completionEvent(context)
        : actionMeta.completionEvent;
    }
  }
  
  return '';
}

  `;
```

```typescript
export interface ParameterAgent {
  actionId: ActionId;
  getAvailableParameters: (context: ActionContext) => Promise<ActionParameter[]>;
  selectBestParameter: (
    availableParameters: ActionParameter[],
    context: ActionContext
  ) => Promise<ActionParameter>;

// EatItem 参数 Agent
export class EatItemParameterAgent implements ParameterAgent {
  actionId = ActionId.Eat_Item;
  
  async getAvailableParameters(context: ActionContext): Promise<ActionParameter[]> {
    const inventory = context.charactorState.inventory || [];
    const availableFood = inventory.filter(item => item.category === 'food');
    
    if (availableFood.length === 0) {
      throw new Error('背包中没有可用的食物');
    }

    return availableFood.map(food => ({
      value: food.name,
      description: `${food.name}可以恢复${food.stamina}点体力`,
      metadata: {
        stamina: food.stamina,
        category: 'food',
        quantity: food.quantity || 1
      }
    }));
  }
  
  async selectBestParameter(
    availableParameters: ActionParameter[],
    context: ActionContext
  ): Promise<ActionParameter> {
    // 如果只有一个选择，直接返回
    if (availableParameters.length === 1) {
      return availableParameters[0];
    }
    
    // 多个选择时，调用 LLM 进行智能选择
    const prompt = `
你是悠酱，现在要选择吃什么食物。

当前状态：
- 体力：${context.charactorState.stamina}/100
- 时间：${context.worldState.time.format('HH:mm')}
- 位置：${context.charactorState.location.major}

可选食物：
${availableParameters.map((param, index) => 
  `${index + 1}. ${param.description}`
).join('\n')}

请选择最合适的食物。考虑因素：
- 当前体力值（体力越低越需要恢复效果好的食物）
- 时间（不同时间适合不同食物）
- 食物的恢复效果

返回选择的食物索引（从1开始）和理由。
    `;

    try {
      const { output } = await generateText({
        model: siliconflow.chat('Qwen/Qwen3-8B'),
        output: Output.object({
          schema: z.object({
            selectedIndex: z.number().min(1).max(availableParameters.length).describe('选择的食物索引（从1开始）'),
            reason: z.string().describe('选择理由')
          })
        }),
        prompt
      });

      const selectedParameter = availableParameters[output.selectedIndex - 1];
      console.log(`EatItem 子 Agent 选择: ${selectedParameter.value}, 理由: ${output.reason}`);
      
      return selectedParameter;
    } catch (error) {
      console.error('EatItem 子 Agent 选择失败，使用默认策略:', error);
      
      // 失败时使用默认策略：选择恢复体力最多的食物
      return availableParameters.reduce((best, current) => {
        const bestStamina = best.metadata?.stamina || 0;
        const currentStamina = current.metadata?.stamina || 0;
        return currentStamina > bestStamina ? current : best;
      });
    }
  }
}
```

### 7.4 主流程协调器

```typescript
// packages/world/src/engine/action-coordinator.ts
export class ActionCoordinator {
  /**
   * 完整的 Action 选择和执行流程
   */
  async executeActionFlow(
    actionList: ActionMetadata[],
    context: ActionContext,
    actionMemoryList: ActionMemory[]
  ): Promise<ExecutionResult> {
    try {
      // 1. 主 Agent 决策阶段
      console.log('🧠 主 Agent 开始决策...');
      const decision = await chooseActionWithTools(actionList, context, actionMemoryList);
      
      console.log(`✅ 主 Agent 选择: ${decision.action}, 理由: ${decision.reason}`);
      
      // 2. 检查是否需要子 Agent 参数选择
      const actionMeta = actionList.find(a => a.action === decision.action);
      if (!actionMeta) {
        throw new Error(`未找到 Action 定义: ${decision.action}`);
      }
      
      let selectedParameters: ActionParameter[] | undefined;
      
      if (actionMeta.parameterAgent) {
        console.log(`🔧 启动子 Agent 进行参数选择: ${decision.action}`);
        
        const parameterAgent = parameterAgents[decision.action];
        if (!parameterAgent) {
          throw new Error(`未找到参数 Agent: ${decision.action}`);
        }
        
        // 2.1 获取可用参数
        const availableParameters = await parameterAgent.getAvailableParameters(context);
        console.log(`📋 可用参数数量: ${availableParameters.length}`);
        
        // 2.2 选择最佳参数
        const selectedParameter = await parameterAgent.selectBestParameter(
          availableParameters,
          context
        );
        
        selectedParameters = [selectedParameter];
        console.log(`✅ 子 Agent 选择参数: ${selectedParameter.value}`);
      }
      
      // 3. 执行阶段
      console.log(`🚀 开始执行 Action: ${decision.action}`);
      const executionResult = await this.executeAction(
        actionMeta,
        context,
        selectedParameters
      );
      
      // 4. 更新计划（如果有）
      if (decision.updateShortTermPlan) {
        await context.charactorState.updateShortTermPlan(decision.updateShortTermPlan);
        console.log('📝 更新短期计划:', decision.updateShortTermPlan);
      }
      
      if (decision.updateLongTermPlan) {
        await context.charactorState.updateLongTermPlan(decision.updateLongTermPlan);
        console.log('📝 更新长期计划:', decision.updateLongTermPlan);
      }
      
      return {
        success: true,
        action: decision.action,
        parameters: selectedParameters,
        result: executionResult,
        reason: decision.reason,
        duration: decision.durationMinute || actionMeta.durationMin
      };
      
    } catch (error) {
      console.error('❌ Action 执行流程失败:', error);
      return {
        success: false,
        error: error.message,
        action: null,
        parameters: null,
        result: null
      };
    }
  }
  
  /**
   * 执行具体的 Action
   */
  private async executeAction(
    actionMeta: ActionMetadata,
    context: ActionContext,
    parameters?: ActionParameter[]
  ): Promise<string> {
    // 设置当前 Action
    await context.charactorState.setAction(actionMeta.action);
    
    // 执行 Action
    await actionMeta.executor(context, parameters);
    
    // 生成完成事件描述
    if (actionMeta.completionEvent) {
      return typeof actionMeta.completionEvent === 'function'
        ? await actionMeta.completionEvent(context, parameters)
        : actionMeta.completionEvent;
    }
    
    return `完成了 ${actionMeta.action}`;
  }
}

// 类型定义
export interface ActionDecision {
  action: ActionId;
  reason: string;
  durationMinute?: number;
  updateShortTermPlan?: string[];
  updateLongTermPlan?: string;
}

export interface ExecutionResult {
  success: boolean;
  action: ActionId | null;
  parameters: ActionParameter[] | null;
  result: string | null;
  reason?: string;
  duration?: number;
  error?: string;
}
```

### 7.5 使用示例

```typescript
// packages/world/src/engine/runner.ts 中的使用
import { ActionCoordinator } from './action-coordinator';

export class SimulationRunner {
  private actionCoordinator = new ActionCoordinator();
  
  async runTick(context: ActionContext) {
    try {
      // 获取可用 Actions
      const availableActions = await this.getAvailableActions(context);
      
      // 获取最近的 Action 记忆
      const recentActions = await this.getRecentActionMemory(context);
      
      // 执行完整的 Action 流程
      const result = await this.actionCoordinator.executeActionFlow(
        availableActions,
        context,
        recentActions
      );
      
      if (result.success) {
        console.log(`✅ Tick 完成: ${result.action} - ${result.result}`);
        
        // 记录 Action 到内存
        await this.recordActionMemory({
          action: result.action!,
          reason: result.reason!,
          parameters: result.parameters,
          timestamp: Date.now(),
          duration: result.duration!
        });
        
        // 推进时间
        await context.worldState.advanceTime(result.duration!);
        
      } else {
        console.error(`❌ Tick 失败: ${result.error}`);
        // 可以实现失败恢复逻辑
      }
      
    } catch (error) {
      console.error('Simulation tick 执行失败:', error);
    }
  }
}
```
